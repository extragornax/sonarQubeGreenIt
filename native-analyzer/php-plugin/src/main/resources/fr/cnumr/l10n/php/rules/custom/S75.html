Compliant Solution - Symfony 

        $query =  $this->createQueryBuilder('c')
            ->select('c')
            ->andWhere('c.user = :userId AND  c.returnedDate IS NULL')
            ->setParameter('userId', $userId)
            ->setMaxResults(10);

        return $query->getQuery()->getResult();
}
